131     /**
132      * Initialize all the components.
133      * @function
134      * @returns {undefined}
135      */
136     this.init = function() {
137         controller = new Controller(this);
138 
139         // layout
140         layout = new dhtmlXLayoutObject(document.body, "3W");
141         layout.cells("a").setWidth(300);
142         layout.cells("a").setText("Database Explorer");
143         layout.cells("b").setWidth(500);
144         layout.cells("b").setText("EntitySet Explorer");
145         layout.cells("c").setText("Entity Viewer");
146 
147         // tree
148         tree = layout.cells("a").attachTree();
149         tree.setImagePath("./imgs/dhxtree_skyblue/");
150         tree.attachEvent("onSelect", fireEventTreeSelected);
151 
152         // grid
153         grid = layout.cells("b").attachGrid();
154         grid.setImagePath("./imgs/dhxgrid_skyblue/");
155         grid.setHeader("_id,_rev");
156         grid.setColAlign("center,center");
157         grid.setColTypes("ro,ro");
158         grid.setColSorting("int,int");
159         grid.enableSmartRendering(true);
160         grid.init();
161         grid.attachEvent("onRowSelect", fireEventRowSelected);
162 
163         // editor
164         editor = layout.cells("c").attachEditor();
165 
166         // toolbars
167         toolbar = layout.attachToolbar();
168         toolbar.attachEvent("onClick", fireEventMainToolbar);
169         esToolbar = layout.cells("a").attachToolbar();
170         esToolbar.attachEvent("onClick", fireEventTreeToolbar);
171         eToolbar = layout.cells("b").attachToolbar();
172         eToolbar.attachEvent("onClick", fireEventGridToolbar);
173 
174         if (toolbar) {
175             toolbar.loadStruct([
176                 {id: "text", type: "text", text: "ODB API Client"},
177                 {id: "sep0", type: "separator"},
178                 {id: "connect", type: "button", text: "Connect To Server"},
179                 {id: "disconnect", type: "button", text: "Disconnect", enabled: "false"},
180                 {id: "sep1", type: "separator"},
181                 {id: "addDB", type: "button", text: "Add Database", enabled: "false"},
182                 {id: "removeDB", type: "button", text: "Remove Database", enabled: "false"},
183                 {id: "sep2", type: "separator"},
184                 {id: "accessRights", type: "button", text: "Access Rights", enabled: "false"},
185                 {id: "envMetaData", type: "button", text: "Env MetaData", enabled: "false"}
186             ]);
187         }
188 
189         if (esToolbar) {
190             esToolbar.loadStruct([
191                 {id: "createES", type: "button", text: "Create ES", enabled: "false"},
192                 {id: "deleteES", type: "button", text: "Delete ES", enabled: "false"},
193                 {id: "getDBMetaData", type: "button", text: "DB MetaData", enabled: "false"},
194                 {id: "getESMetaData", type: "button", text: "ES MetaData", enabled: "false"}
195             ]);
196         }
197 
198         if (eToolbar) {
199             eToolbar.loadStruct([
200                 {id: "createE", type: "button", text: "Create Entity", enabled: "false"},
201                 {id: "updateE", type: "button", text: "Update Entity", enabled: "false"},
202                 {id: "deleteE", type: "button", text: "Delete Entity", enabled: "false"}
203             ]);
204         }
205 
206         windows = new dhtmlXWindows();
207 
208         treeNodeNumber = 0;
209         isConnected = false;
210     };

212     /**
213      * 
214      * @function
215      * @param {JSON} json Reload the source tree with databases and entitysets.
216      * @returns {undefined}
217      */
218     this.loadTreeData = function(json) {
219         tree.deleteChildItems(0);
220         tree.loadJSONObject(json);
221     };
222 
223     /**
224      * Properly "disconnect" the current session.
225      * Disable some buttons, close the source tree.
226      * @function
227      * @param {string} url
228      * @returns {undefined}
229      */
230     this.disconnect = function(url) {
231         if (isConnected) {
232             alert("You have been disconnected from " + url);
233 
234             toolbar.forEachItem(function(id) {
235                 toolbar.disableItem(id);
236             });
237             toolbar.enableItem("connect");
238 
239             esToolbar.forEachItem(function(id) {
240                 esToolbar.disableItem(id);
241             });
242 
243             eToolbar.forEachItem(function(id) {
244                 eToolbar.disableItem(id);
245             });
246             
247             tree.deleteChildItems(0);
248             grid.clearAll();
249             editor.setContent("");
250 
251             isConnected = false;
252         }
253     };

286     /**
287      * Handles events from the main toolbar.
288      * @function
289      * @private
290      * @param {type} id ID of the item triggered.
291      * @returns {undefined}
292      */
293     function fireEventMainToolbar(id) {
294         // depending on the id, requests are made
295         if (id === "connect") {
296             connect();
297         } else if (id === "disconnect") {
298             controller.disconnect();
299         } else if (id === "addDB") {
300             addDB();
301         } else if (id === "removeDB") {
302             removeDB();
303         } else if (id === "accessRights") {
304             controller.getAccessRights();
305         } else if (id === "envMetaData") {
306             controller.getEnvMetaData();
307         }
308     }
309 
310     /**
311      * Handles events from the tree toolbar.
312      * @function
313      * @private
314      * @param {type} id ID of the item triggered.
315      * @returns {undefined}
316      */
317     function fireEventTreeToolbar(id) {
318         // depending on the id, requests are made
319         if (id === "createES") {
320             createES();
321         } else if (id === "deleteES") {
322             controller.deleteES(tree.getItemText(tree.getParentId(tree.getSelectedItemId())),
323                     tree.getUserData(tree.getParentId(tree.getSelectedItemId(), "Database-Name")),
324                     tree.getSelectedItemText());
325         } else if (id === "getDBMetaData") {
326             controller.getDBMetaData(tree.getUserData(tree.getSelectedItemId(), "Database-Name"));
327         } else if (id === "getESMetaData") {
328             controller.getESMetaData(tree.getSelectedItemText(),
329                     tree.getUserData(tree.getSelectedItemId(), "Database-Name"));
330         }
331     }

333     /**
334      * Handles events from the grid toolbar.
335      * @function
336      * @private
337      * @param {type} id ID of the item triggered.
338      * @returns {undefined}
339      */
340     function fireEventGridToolbar(id) {
341         // depending on the id, requests are made
342         if (id === "createE") {
343             createE();
344             fireEventTreeSelected(tree.getSelectedItemId());
345         } else if (id === "updateE") {
346             controller.updateE(tree.getItemText(tree.getParentId(tree.getSelectedItemId())),
347                     tree.getUserData(tree.getParentId(tree.getSelectedItemId()), "Database-Name"),
348                     tree.getSelectedItemText(),
349                     grid.cellById(grid.getSelectedRowId(), 0).getValue(),
350                     editor.getContent().replace(/(<.[^(><.)]+>)/g, "").replace(/ |<p>|<\/p>/g, ""));
351             fireEventTreeSelected(tree.getSelectedItemId());
352         } else if (id === "deleteE") {
353             controller.deleteE(tree.getItemText(tree.getParentId(tree.getSelectedItemId())),
354                     tree.getUserData(tree.getParentId(tree.getSelectedItemId()), "Database-Name"),
355                     tree.getSelectedItemText(),
356                     grid.cellById(grid.getSelectedRowId(), 0).getValue(),
357                     "");
358             fireEventTreeSelected(tree.getSelectedItemId());
359         }
360     }

362     /**
363      * handles events from the grid selection.
364      * @function
365      * @private
366      * @param {string} id Id of the item triggered.
367      * @returns {undefined}
368      */
369     function fireEventRowSelected(id) {
370         if (id !== "new") {
371             // if the "New entity" row has not been selected
372             
373             // get the entity selected...
374             controller.getE(tree.getItemText(tree.getParentId(tree.getSelectedItemId())),
375                     tree.getUserData(tree.getParentId(tree.getSelectedItemId()), "Database-Name"),
376                     tree.getSelectedItemText(),
377                     grid.cellById(grid.getSelectedRowId(), 0).getValue());
378             
379             // ...and activate some buttons
380             eToolbar.forEachItem(function(id) {
381                eToolbar.disableItem(id); 
382             });
383             eToolbar.enableItem("updateE");
384             eToolbar.enableItem("deleteE");
385         } else {
386             // if it has been selected
387             
388             // activate buttons related to an entity creation
389             editor.setContent("Type your entity here.");
390             eToolbar.forEachItem(function(id) {
391                eToolbar.disableItem(id); 
392             });
393             eToolbar.enableItem("createE");
394         }
395     }

397     /**
398      * Handles events from the tree selection.
399      * @function
400      * @private
401      * @param {string} id ID of the item triggered.
402      * @returns {undefined}
403      */
404     function fireEventTreeSelected(id) {
405         if (id.substring(0, DB_KEYWORD.length) === DB_KEYWORD) {
406             // if a database is selected
407             
408             toolbar.enableItem("removeDB");
409 
410             esToolbar.forEachItem(function(id) {
411                 esToolbar.disableItem(id);
412             });
413             esToolbar.enableItem("createES");
414             esToolbar.enableItem("getDBMetaData");
415 
416             eToolbar.forEachItem(function(id) {
417                 eToolbar.disableItem(id);
418             });
419             
420             // get all entity sets to display into the tree
421             controller.getAllES(tree.getSelectedItemText(),
422                     tree.getUserData(tree.getSelectedItemId(), "Database-Name"));
423 
424         } else if (id.substring(0, ES_KEYWORD.length) === ES_KEYWORD) {
425             // if an entityset is selected
426             
427             toolbar.disableItem("removeDB");
428 
429             esToolbar.forEachItem(function(id) {
430                 esToolbar.disableItem(id);
431             });
432             esToolbar.enableItem("deleteES");
433             esToolbar.enableItem("getESMetaData");
434 
435             eToolbar.forEachItem(function(id) {
436                 eToolbar.disableItem(id);
437             });
438                 
439             grid.clearAll();
440             
441             // get all entities to display into the grid
442             controller.getAllE(tree.getItemText(tree.getParentId(id)),
443                     tree.getUserData(tree.getParentId(id), "Database-Name"),
444                     tree.getSelectedItemText());
445         }
446     }

448     /**
449      * Create a popup windows asking for the server address.
450      * @function
451      * @private
452      * @returns {undefined}
453      */
454     function connect() {
455         // create a windo...
456         var window = windows.createWindow("window", 400, 200, 300, 150);
457         window.denyResize();
458         window.setText("Connect to Server");
459 
460         // ...with a form
461         var form = window.attachForm(
462                 [
463                     {type: "settings", position: "label-left", labelWidth: 100, inputWidth: 120},
464                     {type: "block", inputWidth: "auto", offsetTop: 12, list: [
465                             {name: "field", type: "input", label: "Address", value: "http://localhost:8182/"},
466                             {name: "proceed", type: "button", value: "Proceed", offsetLeft: 80, offsetTop: 14}
467                         ]}
468                 ],
469                 true);
470 
471         form.attachEvent("onButtonClick", function() {
472             // pass the server url to the controller
473             controller.connect(form.getItemValue("field"));
474             
475             window.close();
476             isConnected = true;
477             
478             // activate some buttons
479             toolbar.forEachItem(function(id) {
480                 toolbar.enableItem(id);
481             });
482             toolbar.disableItem("connect");
483             toolbar.disableItem("removeDB");
484         });
485     }

487     /**
488      * Create a popup window asking for a databse type and name.
489      * @function
490      * @private
491      * @returns {undefined}
492      */
493     function addDB() {
494         // create a window...
495         var window = windows.createWindow("window", 400, 200, 300, 200);
496         window.denyResize();
497         window.setText("Add Database");
498 
499         // ...with a form
500         var form = window.attachForm(
501                 [
502                     {type: "settings", position: "label-left", labelWidth: 100, inputWidth: 120},
503                     {type: "block", inputWidth: "auto", offsetTop: 12, list: [
504                             {name: "dbType", type: "input", label: "DB Type", value: "couchDB"},
505                             {name: "dbName", type: "input", label: "DB Name", value: "world"},
506                             {name: "proceed", type: "button", value: "Proceed", offsetLeft: 80, offsetTop: 14}
507                         ]}
508                 ],
509                 true);
510 
511         form.attachEvent("onButtonClick", function() {
512             tree.insertNewChild(0, "db" + ++treeNodeNumber, form.getItemValue("dbType"), 0, 0, 0, 0);
513             tree.setUserData("db" + treeNodeNumber, "Database-Name", form.getItemValue("dbName"));
514 
515             window.close();
516             fireEventTreeSelected(tree.getSelectedItemId());
517         });
518     }

524      * @private
525      * @returns {undefined}

527     function removeDB() {

532     }

569     /**
570      * Create a popup window asking for the entity ID and
571      * initiate a request with the correct parameters.
572      * @function
573      * @private
574      * @returns {undefined}
575      */
576     function createE() {
577         // create a window...
578         var window = windows.createWindow("window", 400, 200, 300, 150);
579         window.denyResize();
580         window.setText("Create Entity");
581 
582         // ...and put a form into it
583         var form = window.attachForm(
584                 [
585                     {type: "settings", position: "label-left", labelWidth: 100, inputWidth: 120},
586                     {type: "block", inputWidth: "auto", offsetTop: 12, list: [
587                             {name: "field", type: "input", label: "Entity ID", value: "1"},
588                             {name: "proceed", type: "button", value: "Proceed", offsetLeft: 80, offsetTop: 14}
589                         ]}
590                 ],
591                 true);
592 
593         form.attachEvent("onButtonClick", function() {
594             var editorContent;
595             
596             // check for a valid JSON
597             try {
598                 editorContent = JSON.stringify(JSON.parse(editor.getContent().replace(/(<.[^(><.)]+>)/g, "").replace(/ |<p>|<\/p>/g, "")));
599             }
600             catch (e) {
601                 alert('Invalid JSON');
602                 return;
603             }
604             
605             // send request
606             controller.createE(tree.getItemText(tree.getParentId(tree.getSelectedItemId())),
607                     tree.getUserData(tree.getParentId(tree.getSelectedItemId()), "Database-Name"),
608                     tree.getSelectedItemText(),
609                     form.getItemValue("field"),
610                     editorContent);
611                     
612             window.close();
613         });
